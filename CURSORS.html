1. Write a PL/SQL block to declare a cursor to fetch all employees from EMP table and display their names.
DECLARE
    CURSOR c_emp IS SELECT ENAME FROM SCOTT.EMP;
    v_ename SCOTT.EMP.ENAME%TYPE;
BEGIN
    OPEN c_emp;
    LOOP
        FETCH c_emp INTO v_ename;
        EXIT WHEN c_emp%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE('Employee Name: ' || v_ename);
    END LOOP;
    CLOSE c_emp;
END;

30. Write a PL/SQL block to process employee records and apply different salary increments based on job using cursor.
DECLARE
    CURSOR c_emp IS SELECT ENAME, JOB, SAL FROM SCOTT.EMP FOR UPDATE;
BEGIN
    FOR rec IN c_emp LOOP
        IF rec.JOB = 'CLERK' THEN
            UPDATE SCOTT.EMP SET SAL = SAL + 500 WHERE CURRENT OF c_emp;
        ELSIF rec.JOB = 'SALESMAN' THEN
            UPDATE SCOTT.EMP SET SAL = SAL + 200 WHERE CURRENT OF c_emp;
        ELSE
            UPDATE SCOTT.EMP SET SAL = SAL + 100 WHERE CURRENT OF c_emp;
        END IF;
    END LOOP;
    COMMIT;
END;
Output:
PL/SQL procedure successfully completed.
