30. Write a PL/SQL block to process employee salaries using LOOP, CONTINUE, and EXIT.
DECLARE
    CURSOR emp_cur IS SELECT ENAME, SAL FROM SCOTT.EMP;
    v_count NUMBER := 0;
BEGIN
    FOR rec IN emp_cur LOOP
        -- Skip salaries below 1500
        IF rec.SAL < 1500 THEN
            CONTINUE;
        END IF;
        
        DBMS_OUTPUT.PUT_LINE(rec.ENAME || ' Salary: ' || rec.SAL);
        v_count := v_count + 1;
        
        -- Exit after printing 3 high salaries
        EXIT WHEN v_count = 3;
    END LOOP;
END;
Output (Sample):
Message

JONES Salary: 2975
BLAKE Salary: 2850

PL/SQL procedure successfully completed.
