1. Write a PL/SQL block to handle NO_DATA_FOUND exception while fetching employee details.
DECLARE
    v_ename VARCHAR2(20);
BEGIN
    SELECT ENAME INTO v_ename FROM SCOTT.EMP WHERE EMPNO = 9999;
    DBMS_OUTPUT.PUT_LINE('Name: ' || v_ename);
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Error: Employee not found.');
END;
Output:
Error: Employee not found.
PL/SQL procedure successfully completed.

17. Write a PL/SQL block to catch an exception and re-raise it using RAISE.
BEGIN
    BEGIN
        v_res := 10/0;
    EXCEPTION
        WHEN ZERO_DIVIDE THEN
            DBMS_OUTPUT.PUT_LINE('Logging div by zero locally...');
            RAISE; -- Re-raise to outer block
    END;
EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Caught re-raised error in outer block.');
END;
Output:
Logging div by zero locally...
Caught re-raised error in outer block.
24. Write a PL/SQL block to simulate bank withdrawal with exception handling.
DECLARE
    v_bal NUMBER := 1000;
    v_amt NUMBER := 1500;
    e_insufficient_funds EXCEPTION;
BEGIN
    IF v_amt > v_bal THEN
        RAISE e_insufficient_funds;
    END IF;
    v_bal := v_bal - v_amt;
EXCEPTION
    WHEN e_insufficient_funds THEN
        DBMS_OUTPUT.PUT_LINE('Error: Insufficient balance to withdraw ' || v_amt);
END;
Output:
Error: Insufficient balance to withdraw 1500
PL/SQL procedure successfully completed.
