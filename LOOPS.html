31. Write a PL/SQL block to simulate ATM withdrawal attempts (3 attempts using loop).
DECLARE
    v_correct_pin NUMBER := 1234;
    v_input_pin NUMBER := 0000; -- Simulate wrong pin
    v_attempts NUMBER := 1;
BEGIN
    LOOP
        IF v_input_pin = v_correct_pin THEN
            DBMS_OUTPUT.PUT_LINE('Access Granted.');
            EXIT;
        ELSE
            DBMS_OUTPUT.PUT_LINE('Attempt ' || v_attempts || ': Wrong PIN.');
        END IF;
        
        EXIT WHEN v_attempts = 3;
        v_attempts := v_attempts + 1;
    END LOOP;
    IF v_attempts = 3 AND v_input_pin != v_correct_pin THEN
        DBMS_OUTPUT.PUT_LINE('Card Blocked.');
    END IF;
END;
Output:
Message
Attempt 1: Wrong PIN.
Attempt 2: Wrong PIN.
Attempt 3: Wrong PIN.
Card Blocked.

PL/SQL procedure successfully completed.
