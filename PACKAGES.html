1. Write a package specification and body to display "Hello Oracle".
-- Specification
CREATE OR REPLACE PACKAGE pkg_hello AS
    PROCEDURE say_hello;
END pkg_hello;
/
-- Body
CREATE OR REPLACE PACKAGE BODY pkg_hello AS
    PROCEDURE say_hello IS
    BEGIN
        DBMS_OUTPUT.PUT_LINE('Hello Oracle');
    END say_hello;
END pkg_hello;
/
-- Execution
EXEC pkg_hello.say_hello;
Output:
Hello Oracle
PL/SQL procedure successfully completed.
2. Create a package with one procedure to display employee details.
CREATE OR REPLACE PACKAGE pkg_emp AS
    PROCEDURE show_details(p_empno NUMBER);
END pkg_emp;
/
CREATE OR REPLACE PACKAGE BODY pkg_emp AS
    PROCEDURE show_details(p_empno NUMBER) IS
        v_name VARCHAR2(20);
        v_sal NUMBER;
    BEGIN
        SELECT ENAME, SAL INTO v_name, v_sal FROM SCOTT.EMP WHERE EMPNO = p_empno;
        DBMS_OUTPUT.PUT_LINE('Name: ' || v_name || ' Salary: ' || v_sal);
    END show_details;
END pkg_emp;
/
EXEC pkg_emp.show_details(7788);
Output:
Name: SCOTT Salary: 3000
PL/SQL procedure successfully completed.

19. Create a package that uses cursor to display all employees.
CREATE OR REPLACE PACKAGE pkg_cur AS
    PROCEDURE list_all;
END pkg_cur;
/
CREATE OR REPLACE PACKAGE BODY pkg_cur AS
    PROCEDURE list_all IS
        CURSOR c_emp IS SELECT ENAME FROM SCOTT.EMP;
    BEGIN
        FOR r IN c_emp LOOP
            DBMS_OUTPUT.PUT_LINE(r.ENAME);
        END LOOP;
    END list_all;
END pkg_cur;
/
EXEC pkg_cur.list_all;

ALLEN

23. Create a package with overloaded procedures (same name, different parameters).
CREATE OR REPLACE PACKAGE pkg_overload AS
    PROCEDURE find(id NUMBER);
    PROCEDURE find(name VARCHAR2);
END pkg_overload;
/
CREATE OR REPLACE PACKAGE BODY pkg_overload AS
    PROCEDURE find(id NUMBER) IS 
        v_n VARCHAR2(20);
    BEGIN 
        SELECT ENAME INTO v_n FROM SCOTT.EMP WHERE EMPNO = id;
        DBMS_OUTPUT.PUT_LINE('Found by ID: ' || v_n); 
    END;
    PROCEDURE find(name VARCHAR2) IS 
        v_i NUMBER;
    BEGIN 
        SELECT EMPNO INTO v_i FROM SCOTT.EMP WHERE ENAME = name;
        DBMS_OUTPUT.PUT_LINE('Found by Name, ID: ' || v_i); 
    END;
END pkg_overload;
/
EXEC pkg_overload.find(7788);
EXEC pkg_overload.find('SCOTT');
Output:
Found by ID: SCOTT
Found by Name, ID: 7788
31. Create a package to simulate banking operations (deposit, withdraw, balance check).
CREATE OR REPLACE PACKAGE pkg_bank AS
    PROCEDURE deposit(amt NUMBER);
    PROCEDURE withdraw(amt NUMBER);
    FUNCTION get_bal RETURN NUMBER;
END pkg_bank;
/
CREATE OR REPLACE PACKAGE BODY pkg_bank AS
    v_balance NUMBER := 1000; -- Initial Balance
    PROCEDURE deposit(amt NUMBER) IS BEGIN v_balance := v_balance + amt; END;
    PROCEDURE withdraw(amt NUMBER) IS BEGIN v_balance := v_balance - amt; END;
    FUNCTION get_bal RETURN NUMBER IS BEGIN RETURN v_balance; END;
END pkg_bank;
/
BEGIN
    pkg_bank.deposit(500);
    pkg_bank.withdraw(200);
    DBMS_OUTPUT.PUT_LINE('Balance: ' || pkg_bank.get_bal);
END;
Output:
Balance: 1300
PL/SQL procedure successfully completed.
32. Create a package that uses exception handling for all operations.
CREATE OR REPLACE PACKAGE pkg_exc AS
    PROCEDURE safe_div(a NUMBER, b NUMBER);
END pkg_exc;
/
CREATE OR REPLACE PACKAGE BODY pkg_exc AS
    PROCEDURE safe_div(a NUMBER, b NUMBER) IS
    BEGIN
        DBMS_OUTPUT.PUT_LINE('Result: ' || (a/b));
    EXCEPTION
        WHEN ZERO_DIVIDE THEN
            DBMS_OUTPUT.PUT_LINE('Error: Cannot divide by zero');
    END safe_div;
END pkg_exc;
/
EXEC pkg_exc.safe_div(10, 0);
Output:
Error: Cannot divide by zero
PL/SQL procedure successfully completed.
