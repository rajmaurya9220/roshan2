22. Write a PL/SQL block that uses all concepts: variable, constant, datatype, and output.
DECLARE
    v_emp_name VARCHAR2(50);
    v_base_sal NUMBER(8,2);
    v_join_date DATE;
    c_bonus_rate CONSTANT NUMBER := 0.20;
    v_total_pay NUMBER(8,2);
BEGIN
    -- Fetching employee info
    SELECT ENAME, SAL, HIREDATE INTO v_emp_name, v_base_sal, v_join_date 
    FROM SCOTT.EMP WHERE EMPNO = 7566;
    
    -- Computation
    v_total_pay := v_base_sal + (v_base_sal * c_bonus_rate);
    
    -- Output formatting
    DBMS_OUTPUT.PUT_LINE('--- Employee Pay Slip ---');
    DBMS_OUTPUT.PUT_LINE('Name: ' || v_emp_name);
    DBMS_OUTPUT.PUT_LINE('Joined: ' || v_join_date);
    DBMS_OUTPUT.PUT_LINE('Base Salary: $' || v_base_sal);
    DBMS_OUTPUT.PUT_LINE('Bonus Rate: ' || (c_bonus_rate * 100) || '%');
    DBMS_OUTPUT.PUT_LINE('Total Payable: $' || v_total_pay);
END;
Output:
Message
--- Employee Pay Slip ---
Name: JONES
Joined: 02-APR-81
Base Salary: $2975
Bonus Rate: 20%
Total Payable: $3570

PL/SQL procedure successfully completed.
