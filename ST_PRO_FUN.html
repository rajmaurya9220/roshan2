31. Write a stored function to return nth highest salary.
CREATE OR REPLACE FUNCTION nth_sal(n NUMBER) RETURN NUMBER AS
    v_sal NUMBER;
BEGIN
    SELECT SAL INTO v_sal FROM (
        SELECT SAL, DENSE_RANK() OVER (ORDER BY SAL DESC) rnk FROM SCOTT.EMP
    ) WHERE rnk = n AND ROWNUM = 1;
    RETURN v_sal;
END;
/
SELECT nth_sal(2) FROM DUAL;
Output:
NTH_SAL(2)
----------
      3000
32. Write a procedure to implement transaction (COMMIT / ROLLBACK).
CREATE OR REPLACE PROCEDURE safe_transfer(id1 NUMBER, id2 NUMBER, amt NUMBER) AS
BEGIN
    UPDATE SCOTT.EMP SET SAL = SAL - amt WHERE EMPNO = id1;
    UPDATE SCOTT.EMP SET SAL = SAL + amt WHERE EMPNO = id2;
    COMMIT;
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        DBMS_OUTPUT.PUT_LINE('Error - Transaction Rolled Back');
END;
/
EXEC safe_transfer(7788, 7902, 500);
Output:
PL/SQL procedure successfully completed.
