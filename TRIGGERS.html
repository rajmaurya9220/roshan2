20. Write a trigger to maintain audit table for INSERT, UPDATE, DELETE.
CREATE OR REPLACE TRIGGER trg_full_audit
AFTER INSERT OR UPDATE OR DELETE ON SCOTT.EMP
FOR EACH ROW
DECLARE
    v_op VARCHAR2(10);
BEGIN
    IF INSERTING THEN v_op := 'INSERT';
    ELSIF UPDATING THEN v_op := 'UPDATE';
    ELSE v_op := 'DELETE';
    END IF;
    -- Assume AUDIT_LOG(operation, username, timestamp) exists
    INSERT INTO AUDIT_LOG VALUES (v_op, USER, SYSDATE);
END;
/
Output:
Trigger TRG_FULL_AUDIT compiled

25. Write a trigger to prevent duplicate employee names.
CREATE OR REPLACE TRIGGER trg_unique_name
BEFORE INSERT OR UPDATE OF ENAME ON SCOTT.EMP
FOR EACH ROW
DECLARE
    v_count NUMBER;
BEGIN
    SELECT COUNT(*) INTO v_count FROM SCOTT.EMP WHERE ENAME = :NEW.ENAME;
    IF v_count > 0 THEN
        RAISE_APPLICATION_ERROR(-20010, 'Employee name must be unique.');
    END IF;
END;
/
Output:
Trigger TRG_UNIQUE_NAME compiled
